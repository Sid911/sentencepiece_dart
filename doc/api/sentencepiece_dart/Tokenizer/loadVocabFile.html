Future<void> loadVocabFile(String vocabAssetPath,
    {String separator = '\n'}) async {
  // loading vocab file
  List<String> vocab =
      (await rootBundle.loadString(vocabAssetPath)).split('\n');
  vocabMap =
      Map.fromIterables(vocab, List.generate(vocab.length, (index) => index));
}
